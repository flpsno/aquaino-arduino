
boolean strContains(char *str, char *sfind) {
  char found = 0;
  char index = 0;
  char len;

  len = strlen(str);

  if (strlen(sfind) > len) {
    return false;
  }

  while (index < len) {
    if (str[index] == sfind[found]) {
      found++;
      if (strlen(sfind) == found) {
        return true;
      }
    }
    else {
      found = 0;
    }
    index++;
  }
  return false;
}

void strClear(char *str, char length) {
  for(int i = 0; i < length; i++) {
    str[i] = 0;
  }
}


bool saoIguais(const char *s1, const char *s2)
{
  for (int i = 0; s1[i] == s2[i]; ++i)
  {
    if (s1[i] == 0)
      return true;
  }
  return false;
}

int getLuz() {
  //Zera a variável Luz
  Luz = 0; 
  //Lê o valor do sensor (LDR ligado ao pino A0) e guarda na variável Luz    
  Luz = analogRead(LRD); 
  // Mapeia o valor retornado e converte para porcentagem
  Luz = map(Luz, 0, 1023, 0, 100);

  return Luz;  
}
